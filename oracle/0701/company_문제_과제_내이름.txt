drop sequence com_seq;

create sequence com_seq
increment by 1
start with 1
minvalue 1
maxvalue 10;


drop table company;

	sabun number primary key,
	name varchar2(12),
	day date,
	position varchar2(10),
	pay number
);

SQL> select rownum, name, position, salary
        from company
        order by salary desc;

select rownum as rank, name, position, salary
from (select name, position, salary
        from company
        order by salary desc);

 RANK NAME                 POSITI     SALARY
---------- -------------------- ------ ----------
         1 민아                 부장         9000
         2 현미                 부장         8000
         3 휘성                 대리         4000
         4 찬열                 과장         3000
         5 조권                 과장         2000
         6 보아                 대리         1000


select rank, name, position, salary
from ( select rownum as rank, name, position, salary
	from (select name, position, salary
        		from company
       	 order by salary desc))
where rank>=2 and rank<=4;

     RANK NAME                 POSITI     SALARY
---------- -------------------- ------ ----------
         2 현미                 부장         8000
         3 휘성                 대리         4000
         4 찬열                 과장         3000


select rank, name, position, salary
